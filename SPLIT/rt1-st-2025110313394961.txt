! Chave para Pula

! --- Phase 2: Transform Set ---

 mode tunnel
exit

! --- ACLs para definir tráfego GRE ---

 permit gre host SUBSTITUIR_IP_WAN_SPLIT host SUBSTITUIR_IP_WAN_ZAGREB
! ACL 102: De Split para Pula
ip access-list extended 102
 permit gre host SUBSTITUIR_IP_WAN_SPLIT host SUBSTITUIR_IP_WAN_PULA

! --- Crypto Map ---
crypto map CROATIA 10 ipsec-isakmp
 set peer SUBSTITUIR_IP_WAN_ZAGREB
 set transform-set TSET
 match address 101
!
crypto map CROATIA 20 ipsec-isakmp
 set peer SUBSTITUIR_IP_WAN_PULA
 set transform-set TSET
 match address 102

! --- Aplicar na Interface WAN ---
interface GigabitEthernet0/0

exit
